69 // asn1 encoding
70 // the below constants are based on the number of bytes in the
71 // micro-ecc curve, which can be found here:
72 // https://github.com/kmackay/micro-ecc/blob/master/curve-specific.inc
73 // the maximum asn signature encoding size is found by taking the
74 // size of the raw signature (the number of bytes in its micro-ecc curve * 2)
75 // and then adding 8, to account for the ASN1.SEQUENCE tlv type and length fields,
76 // the two ASN1.INTEGER tlv type and length fields, and the two potential extra
77 // 0's if the integers of the signature containing a leading 1 bit
78 #define NDN_ASN1_ECDSA_ENCODING_MAX_EXTRA_BYTES 8
79 #define NDN_ASN1_ECDSA_ENCODING_MIN_EXTRA_BYTES 6
80 #define NDN_ASN1_ECDSA_SECP256R1_RAW_SIG_SIZE 64

83 #define NDN_ASN1_ECDSA_SECP192R1_RAW_SIG_SIZE 48

119 static inline int
120 ndn_signature_set_signature(ndn_signature_t* signature, const uint8_t* sig_value, size_t sig_size)
121 {
122  if (sig_size > NDN_SIGNATURE_BUFFER_SIZE)
123  return NDN_OVERSIZE;
124 
125  if (signature->sig_type == NDN_SIG_TYPE_ECDSA_SHA256 && sig_size != 64)
126  return NDN_SEC_WRONG_SIG_SIZE;
127 
128  if (signature->sig_type == NDN_SIG_TYPE_HMAC_SHA256 && sig_size != 32)
129  return NDN_SEC_WRONG_SIG_SIZE;
130 
131  if (signature->sig_type == NDN_SIG_TYPE_DIGEST_SHA256 && sig_size != 32)
132  return NDN_SEC_WRONG_SIG_SIZE;
133 
134  signature->sig_size = sig_size;
135  memcpy(signature->sig_value, sig_value, sig_size);
136  return 0;
137 }

200 static inline uint32_t
201 ndn_signature_info_probe_block_size(const ndn_signature_t* signature)
202 {
203  // signature type
204  uint32_t info_buffer_size = encoder_probe_block_size(TLV_SignatureType, 1);
205 
206  if (signature->enable_KeyLocator > 0) {
207  uint32_t key_name_block_size = ndn_name_probe_block_size(&signature->key_locator_name);
208  info_buffer_size += encoder_probe_block_size(TLV_KeyLocator, key_name_block_size);
209  }
210  if (signature->enable_ValidityPeriod > 0) {
211  uint32_t validity_period_buffer_size = encoder_probe_block_size(TLV_NotBefore, 15);
212  validity_period_buffer_size += encoder_probe_block_size(TLV_NotAfter, 15);
213  info_buffer_size += encoder_probe_block_size(TLV_ValidityPeriod, validity_period_buffer_size);
214  }
215  if (signature->enable_SignatureInfoNonce > 0) {
216  info_buffer_size += encoder_probe_block_size(TLV_Nonce, 4);
217  }
218  if (signature->enable_Timestamp > 0) {
219  info_buffer_size += encoder_probe_block_size(TLV_SignedInterestTimestamp,
220  encoder_probe_uint_length(signature->timestamp));
221  }
222  return encoder_probe_block_size(TLV_SignatureInfo, info_buffer_size);
223 }
